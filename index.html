QUESTION 1. 

Create one function with zero parameter having a console statement.

SOLUTION. 
function counter(){
        var counter = 0;
    
        function IncreaseCounter() {
            return counter += 1;
        };
    
        return IncreaseCounter;
    }
    
    var counter = counter();
    console.log(counter());
    console.log(counter());
    console.log(counter());
    console.log(counter());
Output:  
        1
        2
        3
        4

QUESTION 4. 

Write a code to calculate area of a rectangle using inner function. In this case outer function should accept parameter length and inner function should accept parameter breadth.

SOLUTION. 
       function outer(length) {
        function inner(breadth){
          return length*breadth;
        }
        return inner;
       }
       const length=5, breadth=10;
       const area= outer(length)(breadth);
       console.log(area);
Input: length=5 breadth=10

Output: 50
